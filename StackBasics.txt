class Stack():
    def __init__(self):
        self.items = []
    def push(self, item):
        self.items.append(item)
    def pop(self):
        self.items.pop()
    def getStack():
        return self.items
    def peekStack(self):
        if self.items == []:
            print("Error, stack has no elements!")
        else 
            print(self.items[0])
stack = Stack()
stack.push("A")
stack.push("B")
stack.push("C")
print(stack.getStack())
